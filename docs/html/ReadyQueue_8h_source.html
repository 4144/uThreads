26 class ReadyQueue {
27  friend class Cluster;
28 private:
29  std::mutex mtx;//mutex to protect the queue
30  volatile unsigned int size;//keep track of the size of the queue
31  /*
32  * The main producer-consumer queue
33  * to keep track of uThreads in the Cluster
34  */
35  IntrusiveList<uThread> queue;
36  /*
37  * The goal is to use a LIFO ordering
38  * for kThreads, so the queue can self-adjust
39  * itself based on the workload. Thus, a stack
40  * is used to keep track of kThreads blocked on
41  * an empty queue.
42  */
43  IntrusiveList<kThread> ktStack;
44  /*
45  * This variable is used to keep
46  * a history of the latest number
47  * of uThreads popped from the queue.
48  * This will be used to calculate the
49  * new number.
50  */
51  size_t lastPopNum;
52 
53  ReadyQueue() : size(0), lastPopNum(0) {};
54  virtual ~ReadyQueue() {};
55 
56  /*
57  * Remove multiple uThreads from the queue
58  * all at once. 'popnum' is calcluated
59  * everytime based on the last popnum, size
60  * of the queue and number of kThreads in
61  * the cluster.
62  * All uThreads are directly moved to the
63  * local queue of the kThread (nqueue) passed
64  * to the function in bulk. This saves a lot
65  * of overhead.
66  */
67  ssize_t removeMany(IntrusiveList<uThread> &nqueue){
68  //TODO: is 1 (fall back to one task per each call) is a good number or should we used size%numkt
69  //To avoid emptying the queue and not leaving enough work for other kThreads only move a portion of the queue
70  size_t numkt = kThread::currentKT->localCluster->getNumberOfkThreads();
71  size_t popnum = (size / numkt) ? (size / numkt) : 1;
72  popnum = (popnum < lastPopNum)? lastPopNum : popnum;
73  popnum = (popnum > size) ? size : popnum;
74  lastPopNum = popnum;
75 
76  uThread* ut;
77  ut = queue.front();
78  nqueue.transferFrom(queue, popnum);
79  size -= popnum;
80  return popnum;
81  }
82 
83  /*
84  * Unblock a kThread that is waiting for
85  * uThreads to arrive. Each kThread has its
86  * own Condition Variable, and since kThreads
87  * are lining up in LIFO order in ktStack, thus
88  * this code pops a kThread and unblock it through
89  * sending a notificiation on its CV.
90  */
91  inline void unBlock(){
92  if(!ktStack.empty()){
93  kThread* kt = ktStack.back();
94  ktStack.pop_back();
95  kt->cv_flag = true;
96  kt->cv.notify_one();
97  }
98  }
99  /*
100  * This function uses std::mutex to exponentially spin
101  * to provide a semi-spinlock behavior. If after the
102  * first round of exponential spin the lock is not still
103  * available, simply block. This is to avoid overhead of
104  * blocking, although it adds some overhead to cache misses
105  * happening by accessing the mutex memory address.
106  * It can also cause out of ordeing accesses to the mutex (not that
107  * it was fully FIFO before), which is somehow desired to provide
108  * a chance for producers not to be preceded by many consumers that
109  * just need to check whether the queue is empty and block on that.
110  */
111  inline void spinLock(std::unique_lock<std::mutex>& mlock){
112  const uint32_t SPIN_START = 4;
113  const uint32_t SPIN_END = 4 * 1024;
114 
115  size_t spin = SPIN_START;
116  for(;;){
117  if(mlock.try_lock()) break;
118  //exponential spin
119  for ( int i = 0; i < spin; i += 1 ) {
120  asm volatile( "pause" );
121  }
122  spin += spin; // powers of 2
123  if ( spin > SPIN_END ) {
124  //spin = SPIN_START; // prevent overflow
125  mlock.lock();
126  break;
127  }
128  }
129  }
130  /*
131  * Try popping one item and do not block.
132  * give up immediately if cannot acquire
133  * the mutex or the queue is empty.
134  */
135 
136  uThread* tryPop() { //Try to pop one item, or return null
137  uThread* ut = nullptr;
138  /*
139  * Do not block on the lock,
140  * return immediately.
141  */
142  std::unique_lock<std::mutex> mlock(mtx, std::try_to_lock);
143  // no uThreads, or could not acquire the lock
144  if (mlock.owns_lock() && size != 0) {
145  ut = queue.front();
146  queue.pop_front();
147  size--;
148  }
149  return ut;
150  }

152  /*
153  * Try popping multiple items and do not block.
154  * give up immediately if cannot acquire
155  * the mutex or the queue is empty.
156  */
157  ssize_t tryPopMany(IntrusiveList<uThread> &nqueue) {
158  std::unique_lock<std::mutex> mlock(mtx, std::try_to_lock);
159  if(!mlock.owns_lock() || size == 0) return -1;
160  return removeMany(nqueue);
161  }

163  /*
164  * Pop multiple items from the queue and block
165  * if the queue is empty.
166  */
167  ssize_t popMany(IntrusiveList<uThread> &nqueue) {//Pop with condition variable
168  /*
169  * Spin before blocking to avoid the
170  * system call overhead of cv wait on
171  * the consumer and cv wake up on the
172  * producer.
173  * TODO: This was added before spin-block
174  * functionality of mutex, might not be
175  * needed anymore.
176  */
177  for (int spin = 1; spin < 52 * 1024; spin++) {
178  if (size > 0) break;
179  asm volatile("pause");
180  }
181 
182  std::unique_lock<std::mutex> mlock(mtx, std::defer_lock);
183  spinLock(mlock);
184  if (fastpath(size == 0)) {
185  //Push the kThread to stack before waiting on it's cv
186  ktStack.push_back(*kThread::currentKT);
187  /*
188  * Set the cv_flag so we can identify
189  * spurious wake ups from notifies
190  */
191  kThread::currentKT->cv_flag = false;
192  while (size == 0 ) {
193  /*
194  * cv_flag can be true if the kThread were unblocked
195  * by another thread but by the time it acquires the
196  * mutex the queue is empty again. If so the kThread
197  * has been removed from the stack, thus put it back
198  * on the stack
199  */
200  if(kThread::currentKT->cv_flag){
201  ktStack.push_back(*kThread::currentKT);
202  }
203  kThread::currentKT->cv.wait(mlock);
204  }
205  /*
206  * If cv_flag is not true, it means the kThread
207  * experienced a spurious wake up and thus
208  * has not been removed from the stack. Since
209  * ktStack is intrusive the kThread can remove
210  * itself from the stack.
211  */
212  if(!kThread::currentKT->cv_flag){
213  ktStack.remove(*kThread::currentKT);
214  }
215  }
216  ssize_t res = removeMany(nqueue);
217  /*
218  * Each kThread unblocks the next kThread in case
219  * PushMany was called. Only one kThread can always hold
220  * the lock, so there is no benefit in unblocking all kThreads just
221  * for them to be blocked by the mutex.
222  * Chaining the unblocking has the benefit of distributing the cost
223  * of multiple cv.notify_one() calls over producer + waiting consumers.
224  */
225  if(size != 0) unBlock();
226 
227  return res;
228  }

230  /*
231  * Push a single uThread to the queue and
232  * wake one kThread up.
233  */
234  void push(uThread* ut) {
235  std::unique_lock<std::mutex> mlock(mtx, std::defer_lock);
236  spinLock(mlock);
237  queue.push_back(*ut);
238  size++;
239  unBlock();
240  }
241 
242  /*
243  * Push multiple uThreads to the queue, and wake one
244  * kThread up. That kThread then wakes up another kThread
245  * if there are still uThreads left in the queue.
246  * uThreads are copied directly from the passed container
247  * to the queue in bulk to avoid the overhead.
248  */
249  void pushMany(IntrusiveList<uThread>& utList, size_t count){
250  std::unique_lock<std::mutex> mlock(mtx, std::defer_lock);
251  spinLock(mlock);
252  queue.transferFrom(utList, count);
253  size+=count;
254  unBlock();
255  }
256 
257  /*
258  * Whether the queue is empty or not.
259  */
260  bool empty() const {
261  return queue.empty();
262  }
